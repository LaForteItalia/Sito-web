import express from 'express'; import cors from 'cors'; import axios from 'axios'; const app = express(); const port = 3000; // =============================== // CONFIGURAZIONE SERVER // =============================== const SERVER_ADDRESS = '45.138.201.37:30120'; const CFX_ID = '4odj5q'; app.use(cors()); app.use(express.json()); // =============================== // STATUS ENDPOINT // =============================== app.get('/status', async (req, res) => { try { console.log(`================================================`); console.log(`[${new Date().toLocaleTimeString()}] Controllo FiveM`); // Recupera players.json const playersResponse = await axios.get( `http://${SERVER_ADDRESS}/players.json`, { timeout: 5000 } ); // Recupera info.json const infoResponse = await axios.get( `http://${SERVER_ADDRESS}/info.json`, { timeout: 5000 } ); console.log('players.json:', playersResponse.data); const playerCount = Array.isArray(playersResponse.data) ? playersResponse.data.length : 0; const info = infoResponse.data || {}; const serverName = info.vars?.sv_projectName || info.vars?.sv_hostname || 'FiveM Server'; const maxPlayers = Number(info.vars?.sv_maxClients) || 0; res.json({ online: true, players: playerCount, maxPlayers: maxPlayers, serverName: serverName, ip: SERVER_ADDRESS, cfx: CFX_ID, timestamp: Date.now() }); } catch (err) { console.error('Errore:', err.message); res.json({ online: false, players: 0, maxPlayers: 0, serverName: 'Offline', ip: SERVER_ADDRESS, cfx: CFX_ID, timestamp: Date.now() }); } }); // =============================== // HOME // =============================== app.get('/', (req, res) => { res.send('Backend FiveM attivo ✅'); }); // =============================== // START SERVER // =============================== app.listen(port, '0.0.0.0', () => { console.log(`================================================`); console.log(`🚀 BACKEND FIVEM ONLINE`); console.log(`🌐 http://localhost:${port}/status`); console.log(`🎮 ${SERVER_ADDRESS}`); console.log(`================================================`); });
